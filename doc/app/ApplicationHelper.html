    # File app/helpers/application_helper.rb, line 74
def link_to_add_fields(inner_text, f, association, category_id, new_object)
        fields = f.fields_for(association, new_object, :child_index => "new_#{association}") do |builder|
                render(association.to_s.singularize + "_fields", :f => builder)
        end
        link_to_function(inner_text, "add_#{association.to_s.singularize}_fields(this, '#{association}', '#{escape_javascript(fields)}')")
end

    sortable(column, title = nil, icon_asc = "icon-arrow-up icon-white", icon_desc = "icon-arrow-down icon-white", icon_default = "icon-list icon-white") click to toggle source

    Creates a link tag that handle the sorting criteria (column name and direction). It passes to the controller the :sort and :direction parameters.

    Arguments:

    • column, identifies the column name used in the sorting criteria

    • title, the description used in the link tag element

    • icon_asc, bootstrap icon css classes used for the ascending icon rappresentation (default: icon-arrow-up icon-white)

    • icon_desc, bootstrap icon css classes used for the descending icon rappresentation (default: icon-arrow-down icon-white)

    • icon_default, bootstrap icon css classes used for the default icon rappresentation (default: icon-list icon-white)

    # File app/helpers/application_helper.rb, line 90
def sortable(column, title = nil, icon_asc = "icon-arrow-up icon-white", icon_desc = "icon-arrow-down icon-white", icon_default = "icon-list icon-white")
        title ||= column.titleize
        css_class = column == sort_column ? "current #{sort_direction}" : nil
        direction = column == sort_column && sort_direction == "asc" ? "desc" : "asc"
        i_css_class = column == sort_column ? sort_direction == "asc" ? icon_asc : icon_desc : icon_default
        
        link_to "#{title} #{content_tag :i, nil, class: i_css_class}".html_safe, {:sort => column, :direction => direction, :q => params[:q].nil? ? '' : params[:q]}, {:class => css_class}
end
