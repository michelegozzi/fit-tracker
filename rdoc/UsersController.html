    # File app/controllers/users_controller.rb, line 65
def destroy
        @user = User.find(params[:id])
        if current_user?(@user)
                flash[:error] = "You can't destroy yourself."
                redirect_to users_url
        else
                @user.destroy
                flash[:success] = "User destroyed."
                redirect_to users_url
        end

end

    show() click to toggle source

    Accepts:

    1. users’s params[:id]

    2. sort_column, see the helper method sort_column

    3. sort_direction

    4. params

    Returns:

    1. an @user instance selected by params[:id]

    2. a sorted collection of sheets, handled by the user, in a paginated form (7 elements per page)

    # File app/controllers/users_controller.rb, line 23
def show
        @user = User.find(params[:id])
        @sheets = @user.sheets.order(sort_column + " " + sort_direction).paginate(page: params[:page], :per_page => 7)
end
