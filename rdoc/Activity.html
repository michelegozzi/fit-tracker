def set_timepart
        logger.debug "DBG_20130619_0816 after_initialize"
        if self.time.nil?
                self.timepart = Time.local(2000, 1, 1).strftime("%I:%M %p")
        else
                self.timepart = self.time.strftime("%I:%M %p")
        end

rescue ActiveRecord::MissingAttributeError
end
